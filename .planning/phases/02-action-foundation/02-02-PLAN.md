   const PLATFORM_MAP = {
     'Linux/X64':   'Linux-x86_64',
     'Linux/ARM64': 'Linux-aarch64',

   function detectPlatform(runner) {
     // runner = { os: process.env.RUNNER_OS, arch: process.env.RUNNER_ARCH }
     const key = `${runner.os}/${runner.arch}`;
     const platformName = PLATFORM_MAP[key];
     if (!platformName) {
       throw new Error(
         `Unsupported platform: ${runner.os}/${runner.arch}. ` +

1. Replace the skeleton src/index.js with the full implementation:
   ```javascript
   const tc = require('@actions/tool-cache');
   const core = require('@actions/core');
   const { detectPlatform } = require('./platform');
   const { resolveLatestVersion, validateVersion } = require('./version');

   const CHUCKD_COMPAT = '1.0';

   async function run() {
     try {
       // 1. Platform detection
       const platformName = detectPlatform({
         os: process.env.RUNNER_OS,
         arch: process.env.RUNNER_ARCH,
       });
       core.info(`Platform: ${platformName}`);

       // 2. Version resolution
       let version = process.env.INPUT_VERSION;
       if (!version) {
         version = await resolveLatestVersion(CHUCKD_COMPAT, process.env.INPUT_GITHUB_TOKEN);
         core.info(`Resolved latest compatible version: ${version}`);
       } else {
         validateVersion(version, CHUCKD_COMPAT);
         core.info(`Using specified version: ${version}`);
       }

       // 3. Check tool cache
       const toolName = 'chuckd';
       let cachedPath = tc.find(toolName, version);

       if (!cachedPath) {
         // 4. Download from GitHub releases
         const tarball = `chuckd-${platformName}-${version}.tar.gz`;
         const downloadUrl = `https://github.com/anentropic/chuckd/releases/download/${version}/${tarball}`;
         core.info(`Downloading chuckd ${version} for ${platformName}`);

         const downloadPath = await tc.downloadTool(
           downloadUrl,
           undefined,
           `token ${process.env.INPUT_GITHUB_TOKEN}`
         );
         const extractedPath = await tc.extractTar(downloadPath);
         cachedPath = await tc.cacheDir(extractedPath, toolName, version);
       } else {
         core.info(`Using cached chuckd ${version}`);
       }

       // 5. Add to PATH
       core.addPath(cachedPath);
       core.info(`chuckd ${version} available on PATH`);

     } catch (error) {
       core.setFailed(error.message);
     }
   }

   module.exports = { run };
   run();
   ```

   Key points:
   - Read inputs from process.env.INPUT_* (NOT core.getInput -- composite actions don't set those)
   - Use tc.find() before downloading (cache check)
   - Auth header format: `token ${GITHUB_TOKEN}` for tc.downloadTool
   - core.setFailed on any error (sets action exit code AND logs error)
   - Export run for testability but also call it at module level
