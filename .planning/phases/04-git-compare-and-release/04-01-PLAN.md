    Add exported function `extractBaseRefSchema(baseRef, schemaFile)`:
    ```javascript
    const execFileAsync = promisify(execFile);

    export async function extractBaseRefSchema(baseRef, schemaFile) {
      // Step 1: Detect shallow clone
      const { stdout: shallowOut } = await execFileAsync('git', ['rev-parse', '--is-shallow-repository']);
      if (shallowOut.trim() === 'true') {
        throw new Error(
          'Shallow clone detected. Set fetch-depth: 0 in your actions/checkout step:\n' +
          '  - uses: actions/checkout@v4\n' +
          '    with:\n' +
          '      fetch-depth: 0'
        );
      }

      // Step 2: Extract file content from base ref
      const gitRef = `${baseRef}:${schemaFile}`;
      let fileContent;
      try {
        const result = await execFileAsync('git', ['show', gitRef]);
        fileContent = result.stdout;
      } catch (err) {
        throw new Error(`Failed to read ${schemaFile} from ref '${baseRef}': ${err.stderr || err.message}`);
      }

      // Step 3: Write to temp file
      const tmpDir = process.env.RUNNER_TEMP || os.tmpdir();
      const ext = path.extname(schemaFile);
      const basename = path.basename(schemaFile, ext);
      const tmpPath = path.join(tmpDir, `chuckd-base-${basename}-${Date.now()}${ext}`);
      await writeFile(tmpPath, fileContent);
      return tmpPath;
    }
    ```
