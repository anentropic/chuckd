    Create `.github/workflows/release.yml` with exactly this content:

    ```yaml
    name: Release

    on:
      release:
        types: [published]

    permissions:
      contents: write

    jobs:
      update-major-tag:
        runs-on: ubuntu-latest
        steps:
          - uses: actions/checkout@v4
          - name: Move major version tag
            run: |
              VERSION=${GITHUB_REF#refs/tags/}
              MAJOR=${VERSION%%.*}
              git config user.name "github-actions[bot]"
              git config user.email "github-actions[bot]@users.noreply.github.com"
              git tag -fa "${MAJOR}" -m "Update ${MAJOR} to ${VERSION}"
              git push origin "${MAJOR}" --force
    ```

    This workflow triggers on every GitHub release publish event. It extracts the major version from the tag (e.g., `v1.2.3` -> `v1`), force-moves the floating major tag to point at the same commit, and pushes it. The `permissions: contents: write` is required for tag push.

    ```yaml
    name: Integration Tests

    on:
      push:
      pull_request:
      workflow_dispatch:

    jobs:
      test:
        strategy:
          fail-fast: false
          matrix:
            runner: [ubuntu-24.04, ubuntu-24.04-arm, macos-15]
        runs-on: ${{ matrix.runner }}

        steps:
          - uses: actions/checkout@v4
            with:
              fetch-depth: 0

          # Test 1: Explicit-path mode, compatible (should succeed)
          - name: "explicit-path: compatible schemas"
            uses: anentropic/chuckd-action@structured-output
            with:
              schema-file: schemas/person.v2.json
              previous-schemas: schemas/person.v1.json

          # Test 2: Explicit-path mode, incompatible (should fail)
          - name: "explicit-path: incompatible schemas"
            id: explicit-incompatible
            uses: anentropic/chuckd-action@structured-output
            continue-on-error: true
            with:
              schema-file: schemas/person.v2-breaking.json
              previous-schemas: schemas/person.v1.json

          - name: "assert explicit-incompatible failed"
            if: steps.explicit-incompatible.outcome != 'failure'
            run: |
              echo "Expected explicit-path incompatible check to fail, but got: ${{ steps.explicit-incompatible.outcome }}"
              exit 1

          # Test 3: Git-compare mode, compatible (should succeed)
          # Compares schemas/person.json on HEAD (v2 compatible) against baseline tag (v1)
          - name: "git-compare: compatible schema"
            uses: anentropic/chuckd-action@structured-output
    ```
